91. Decode Ways

226

class Solution(object):
    def numDecodings(self, s):
        """
        :type s: str
        :rtype: int
        """

        if not s:
            return 0

        dp = [0] * (len(s)+1)

        dp[0] = 1

        if s[0] != '0':
            dp[1] = 1


        for i in range(2,len(s)):
            digits = int(s[i-1,i])
            tens = int(s[i-2,i])

            if digits >= 1 and digits <= 9:
                dp[i] += dp[i-1]
            if tens >= 10 and tens <= 26:
                dp[i] += dp[i-2]

235. Lowest Common Ancestor of a Binary Search Tree 

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def lowestCommonAncestor(self, root, p, q):
        """
        :type root: TreeNode
        :type p: TreeNode
        :type q: TreeNode
        :rtype: TreeNode
        """

        if not root:
            return root

        if p.val > q.val:
            return self.dfs(root,q,p)

        return self.dfs(root,p,q)

    def dfs(self,root,p,q):

        if root.val >= p.val and root.val <= q.val:
            return root
        elif root.val > p.val and root.val > q.val:
            return self.dfs(root.left,p,q)
        else:
            return self.dfs(root.right,p,q)

33. Search in Rotated Sorted Array 

class Solution(object):
    def search(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: int
        """
        i = 0
        j = len(nums) - 1

        while i <= j:

            mid = i + (j-i) // 2

            if nums[mid] == target:
                return mid

            if nums[j] > nums[mid]:

                if target > nums[mid] and target <= nums[j]:
                    i = mid + 1
                else:
                    j = mid -1

            elif nums[j] < nums[mid]:

                if target < nums[mid] and target >= nums[i]:
                    j = mid - 1
                else:
                    i = mid + 1

            else:
                i = mid + 1


        return -1
