91. Decode Ways

226

class Solution(object):
    def numDecodings(self, s):
        """
        :type s: str
        :rtype: int
        """

        if not s:
            return 0

        dp = [0] * (len(s)+1)

        dp[0] = 1

        if s[0] != '0':
            dp[1] = 1


        for i in range(2,len(s)):
            digits = int(s[i-1,i])
            tens = int(s[i-2,i])

            if digits >= 1 and digits <= 9:
                dp[i] += dp[i-1]
            if tens >= 10 and tens <= 26:
                dp[i] += dp[i-2]
