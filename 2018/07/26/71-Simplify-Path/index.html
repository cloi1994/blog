71. Simplify Path 

class Solution(object):
    def simplifyPath(self, path):
        """
        :type path: str
        :rtype: str
        """

        path = path.split('/')

        res = ""

        jump = 0

        for i in range(len(path)-1,-1,-1):

            if path[i] == '' or path[i] =='.':
                continue
            if path[i] == '..':
                jump += 1
                continue
            if jump == 0:
                res =  '/' + path[i]  + res
            else:
                jump -= 1

        if res == "":
            return '/'

        return res
