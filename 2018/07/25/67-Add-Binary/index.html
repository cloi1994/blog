67. Add Binary 

class Solution(object):
    def addBinary(self, a, b):
        """
        :type a: str
        :type b: str
        :rtype: str
        """
        i = len(a) - 1
        j = len(b) - 1

        res = ""

        carry = 0

        while i >= 0 or j >= 0 or carry:

            cur = carry

            if i >= 0:
                cur += int(a[i])
                i -= 1
            if j >= 0:
                cur += int(b[j])
                j -= 1

            carry = cur / 2
            res = str(cur % 2) + res

        return res
