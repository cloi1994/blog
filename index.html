234. Palindrome Linked List

if odd -> pop the stack
if even -> no pop
move slow pointer to next place

# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def isPalindrome(self, head):
        """
        :type head: ListNode
        :rtype: bool
        """

        if not head:
            return True

        stack = [head.val]

        slow = head
        fast = head

        while fast.next and fast.next.next:

            slow = slow.next
            fast = fast.next.next
            stack.append(slow.val)

        if not fast.next:
            stack.pop()

        slow = slow.next

        while slow:

            if stack == []:
                return False
            if stack.pop() != slow.val:
                return False

            slow = slow.next

        return True


    def reverse(self,head):

        prev = None

        cur = head

        while cur:

            curNext = cur.next
            cur.next = prev
            prev = cur
            cur = curNext

        return prev

57. Insert Interval

Solution # 1

skip the interval that doesn’t overlap to the new interval,
insert the new interval to the res and merge it to the rest

Solution 2

Do a BinarySearch on left bound and right bound

def insert2(self, intervals, newInterval):
    if not intervals:
        return [newInterval]
    elif newInterval.end < intervals[0].start:
        return [newInterval] + intervals
    elif newInterval.start > intervals[-1].end:
        return intervals + [newInterval]

    def leftSearch():
        target = newInterval.start
        lo = 0
        hi = len(intervals) - 1
        if newInterval.start < intervals[lo].start:
            return lo

        while lo <= hi:
            if lo == hi:
                return lo
            mid = (lo + hi)//2
            if intervals[mid].start <= target <= intervals[mid].end:
                return mid
            # [1,5] ,[7,9]
            # insert [6,10]
            if mid > 0 and intervals[mid-1].end < target <= intervals[mid].start:
                return mid

            elif intervals[mid].start > target:
                hi = mid - 1
            else:
                lo = mid + 1
        # Should exit by now

    # Above right is first non-overlapping
    # But in this solution, right is last overlapping
    def rightSearch():
        target = newInterval.end
        lo = 0
        hi = len(intervals) - 1
        if newInterval.end > intervals[hi].end:
            return hi

        while lo <= hi:
            if lo == hi:
                return hi
            mid = (lo + hi) // 2
            if intervals[mid].start <= target <= intervals[mid].end:
                return mid
            # [1,5][6,9]
            # insert [2,5] 
            if mid < len(intervals) - 1 and intervals[mid].end <= target < intervals[mid+1].start:
                return mid
            elif intervals[mid].start > target:
                hi = mid - 1
            else:
                lo = mid + 1
        # Should exit by now

    left = leftSearch()
    right = rightSearch()

    newInterval.start = min(newInterval.start, intervals[left].start)
    newInterval.end = max(newInterval.end, intervals[right].end)

    return intervals[:left] + [newInterval] + intervals[right+1:]

325. Maximum Size Subarray Sum Equals k

nums: [-2, -1, 2, 1], k = 1

sums: [-2, -3, -1, 0]

我们发现累积和中没有数字等于k，但是我们知道这个例子的答案是[-1, 2]，
那么我们看累积和数组的第一和第三个数字，我们是否能看出一些规律呢，
没错，第三个数字-1减去k，得到第一个数字，这就是规律，这也是累积和求区间和的方法，
但是由于累计和数组中可能会有重复数字，而哈希表的关键字不能相同，

cur_sum - k in hm == [i:j] inclusive

i = hm[cur_sum - k]

class Solution:
    """
    @param nums: an array
    @param k: a target value
    @return: the maximum length of a subarray that sums to k
    """
    def maxSubArrayLen(self, nums, k):
            hm = {}
            maxSize = 0
            summ = 0

            for i in range(len(nums)):
                summ += nums[i]

                #if up to cur position summ = k
                if summ == k:
                    maxSize = i + 1
                    continue
                #if pervious index sum up to cur
                if summ - k in hm:
                    maxSize = max(i-hm[summ-k],maxSize)

                if summ not in hm:
                    hm[summ] = i

            return maxSize
